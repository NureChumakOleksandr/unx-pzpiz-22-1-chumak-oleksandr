3 ОПИС ВИКОНАНОЇ РОБОТИ

	У ході виконання лабораторної роботи № 2 було розроблено Bash-скрипт для збору та обробки системної інформації у середовищі Unix/Linux. Скрипт реалізує наступну функціональність:
    1. Обробка аргументів командного рядка 
        ◦ Підтримується опція -h|--help для виведення довідки. 
        ◦ Параметр -n NUM дозволяє обмежити кількість архівних копій файлу. 
        ◦ Можливе вказання власного шляху до файлу для запису результатів. 
    2. Локалізація повідомлень 
        ◦ Якщо системна локаль uk_UA, повідомлення виводяться українською мовою; інакше — англійською. 
    3. Збір системної інформації 
        ◦ Hardware: CPU, обсяг RAM, виробник і модель материнської плати, серійний номер системи. 
        ◦ System: дистрибутив ОС, версія ядра, дата встановлення системи, ім’я хоста, час роботи системи, кількість процесів і користувачів. 
        ◦ Network: інформація про всі мережеві інтерфейси з IP-адресами або позначкою -/- у разі їх відсутності. 
    4. Форматування виводу 
        ◦ Дотримано стандартизовану структуру розділів Hardware, System і Network. 
        ◦ Вивід містить поточну дату у форматі Date: ... та Unix Timestamp одного моменту часу. 
        ◦ Використовується точне форматування з лапками та роздільниками. 
    5. Робота з файлами 
        ◦ У разі відсутності каталогу для логів, він створюється за допомогою mkdir -p. 
        ◦ Якщо файл вже існує, старі файли перейменовуються у формат ім’я-YYYYMMDD-NNNN. 
        ◦ Кількість архівів контролюється параметром -n, старіші файли автоматично видаляються. 
    6. Обробка помилок і логування 
        ◦ Некоректні параметри викликають повідомлення у stderr та завершення скрипта з кодом ≠ 0. 
        ◦ Відсутні або недоступні дані позначаються як Unknown. 
        ◦ Використано функції для підвищення модульності та підтримки розширюваності скрипта. 
	Таким чином, скрипт забезпечує надійний збір і архівацію системної інформації з дотриманням вимог лабораторної роботи.

4 ВИСНОВКИ

	В ході виконання лабораторної роботи № 2 було розроблено Bash-скрипт для збору та обробки системної інформації у середовищі Unix/Linux, який коректно збирає дані про апаратне забезпечення, операційну систему та мережеві інтерфейси, виводить їх у стандартизованому форматі, одночасно записує у лог-файл, забезпечує архівацію та контроль кількості збережених копій, обробляє помилки та підтримує локалізацію повідомлень, що дозволило набути практичних навичок роботи з утилітами системного рівня, форматуванням виводу та управлінням файлами в Unix/Linux.

   110	cleanup_archives() {
   111	    local dir="$1"
   112	    local base="$2"
   113	    local keep="$3"
   114	
   115	    if [ -z "$keep" ]; then
   116	        return
   117	    fi
   118	
   119	    local archives=()
   120	    for f in "$dir/${base}-"[0-9][0-9][0-9][0-9][0-9][0-9][0-9][0-9]-[0-9][0-9][0-9][0-9]; do
   121	        if [ -e "$f" ]; then
   122	            archives+=("$f")
   123	        fi
   124	    done
   125	
   126	    IFS=$'\n' sorted=($(printf '%s\n' "${archives[@]}" | sort)); unset IFS
   127	
   128	    local total=${#sorted[@]}
   129	    if [ "$total" -gt "$keep" ]; then
   130	        local to_delete=$((total - keep))
   131	        for ((i = 0; i < to_delete; i++)); do
   132	            rm -f "${sorted[$i]}"
   133	        done
   134	    fi

   137	rename_existing() {
   138	    get_next_number() {
   139	        local dir="$1"
   140	        local base="$2"
   141	        local today="$3"
   142	        local max=-1
   143	
   144	        for f in "$dir/${base}-${today}-"[0-9][0-9][0-9][0-9]; do
   145	            if [ -e "$f" ]; then
   146	                local num_part
   147	                num_part=$(basename "$f" | grep -oP '\d{4}$')
   148	                local num_val=$((10#$num_part))
   149	                if [ "$num_val" -gt "$max" ]; then
   150	                    max=$num_val
   151	                fi
   152	            fi
   153	        done
   154	
   155	        printf "%04d" $((max + 1))
   156	    }
   157	
   158	    local filepath="$1"
   159	    local dir
   160	    local base
   161	    dir="$(dirname "$filepath")"
   162	    base="$(basename "$filepath")"
   163	
   164	    local today
   165	    today="$(date '+%Y%m%d')"

   167	    local next_num
   168	    next_num="$(get_next_number "$dir" "$base" "$today")"
   169	
   170	    NEW_ARCHIVE_NAME="${base}-${today}-${next_num}"
   171	    local new_path="${dir}/${NEW_ARCHIVE_NAME}"
   172	
   173	    mv "$filepath" "$new_path"

   176	create_and_write() {
   177	    get_sys_spec_overview() {
   178	        get_cpu() {
   179	            local cpu=$(lscpu 2>/dev/null | grep 'Model name' | awk -F: '{print $2}' | xargs)
   180	            [[ -z "$cpu" ]] && cpu="Unknown"
   181	            echo "$cpu"
   182	        }
   183	
   184	        get_ram() {
   185	            local ram=$(grep MemTotal /proc/meminfo | awk '{print int($2/1024)}')
   186	            [[ -z "$ram" ]] && ram="Unknown"
   187	            echo "$ram"
   188	        }
   189	
   190	        get_motherboard() {
   191	            local board_man=$(sudo dmidecode -s baseboard-manufacturer 2>/dev/null | xargs || echo "Unknown")
   192	            local board_prod=$(sudo dmidecode -s baseboard-product-name 2>/dev/null || echo "Unknown")
   193	            echo "$board_man $board_prod"
   194	        }
   195	
   196	        get_os() {
   197	            local os=$(grep 'PRETTY_NAME' /etc/os-release | awk -F\" '{print $2}')
   198	            [[ -z "$os" ]] && os="Unknown"
   199	            echo "$os"
   200	        }
   201	
   202	        get_install_date() {
   203	            local date=$(stat / 2>/dev/null | grep 'Birth' | awk '{print $2, $3}')
   204	            [[ -z "$date" ]] && date="Unknown"
   205	            echo "$date"
   206	        }

   208	        # Date
   209	        local NOW=$(date "+%s")
   210	        local DATE_STR=$(LC_ALL=C date -d "@$NOW" "+Date: %a, %d %b %Y %H:%M:%S %z")
   211	        local UNIX_TS="Unix Timestamp: $NOW"
   212	
   213	        # Hardware
   214	        local CPU=$(get_cpu)
   215	        local RAM=$(get_ram)
   216	        local MOTHERBOARD=$(get_motherboard)
   217	        local SERIAL=$(sudo dmidecode -s system-serial-number 2>/dev/null || echo "Unknown")
   218	
   219	        # System
   220	        local OS=$(get_os)
   221	        local KERNEL=$(uname -r)
   222	        local INSTALL_DATE=$(get_install_date)
   223	        local HOST=$(hostname)
   224	        local UPTIME=$(uptime -p)
   225	        local PROCESSES=$(ps -A --no-headers | wc -l)
   226	        local USERS=$(who | wc -l)
   227	        local NETWORK=$(ip -o addr | awk '{iface=$2; ip=$4; if(ip=="") ip="-/-"; print iface": "ip}')
   228	
   229	        echo "$DATE_STR"
   230	        echo "$UNIX_TS"
   231	        echo "---- Hardware ----"
   232	        echo "CPU: \"$CPU\""
   233	        echo "RAM: $RAM MB"
   234	        echo "Motherboard: $MOTHERBOARD"
   235	        echo "System Serial Number: $SERIAL"
   236	        echo "---- System ----"
   237	        echo "OS Distribution: \"$OS\""
   238	        echo "Kernel version: $KERNEL"
   239	        echo "Installation date: $INSTALL_DATE"
   240	        echo "Hostname: $HOST"
   241	        echo "Uptime: $UPTIME"
   242	        echo "Processes running: $PROCESSES"
   243	        echo "Users logged in: $USERS"
   244	        echo "---- Network ----"
   245	        echo "$NETWORK"
   246	        echo '----"EOF"----'
   247	    }
   248	
   249	    local filepath="$1"
   250	    local dir
   251	    dir="$(dirname "$filepath")"
   252	
   253	    if [ ! -e "$dir" ]; then
   254	        if ! mkdir -p "$dir" 2>/dev/null; then
   255	            log_error "$MSG_INVALID"
   256	            exit 1
   257	        fi
   258	    fi
   259	
   260	    if ! touch "$filepath" 2>/dev/null; then
   261	        log_error "$MSG_INVALID"
   262	        exit 1
   263	    fi
   264	
   265	    get_sys_spec_overview | tee "$OUTPUT_FILE"
   266	}
